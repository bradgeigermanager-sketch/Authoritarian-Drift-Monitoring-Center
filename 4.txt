# **🔥 2. Full Auto‑Compiler (JavaScript)**  
Drop this into `schema-compiler.js`.

```javascript
async function loadJSONSchemas() {
  const res = await fetch("/api/schemas");
  if (!res.ok) throw new Error("Failed to load JSON schemas");
  return await res.json();
}

async function loadCSV(file) {
  const res = await fetch(`/api/options?file=${encodeURIComponent(file)}`);
  if (!res.ok) throw new Error(`Failed to load CSV: ${file}`);
  return await res.json(); // [{value,label}, ...]
}

async function compileSchemas() {
  const rawSchemas = await loadJSONSchemas();
  const compiled = [];

  for (const schema of rawSchemas) {
    const out = {
      title: schema.title,
      description: schema.description || "",
      fields: []
    };

    for (const field of schema.fields) {
      const f = { ...field };

      // Expand CSV-backed selects/checklists
      if (f.type === "select" && f.source) {
        f.options = await loadCSV(f.source);
      }

      if (f.type === "checklist" && f.source) {
        const csvItems = await loadCSV(f.source);
        f.items = csvItems.map(i => i.label);
      }

      // Normalize radios
      if (f.type === "radio" && f.options) {
        f.options = f.options.map(o => ({
          value: o.value,
          label: o.label
        }));
      }

      // Validate minimal structure
      if (!f.name || !f.label || !f.type) {
        console.warn("Invalid field skipped:", f);
        continue;
      }

      out.fields.push(f);
    }

    compiled.push(out);
  }

  return compiled;
}
```

# **🔥 3. Validation Layer (Optional but recommended)**  
Add this to `schema-compiler.js`.

```javascript
function validateCompiledSchema(schema) {
  const errors = [];

  if (!schema.title) errors.push("Missing title");
  if (!Array.isArray(schema.fields)) errors.push("Fields must be an array");

  schema.fields.forEach(f => {
    if (!f.name) errors.push(`Field missing name: ${f.label}`);
    if (!f.label) errors.push(`Field missing label: ${f.name}`);
    if (!f.type) errors.push(`Field missing type: ${f.name}`);

    if (f.type === "slider") {
      if (f.min === undefined || f.max === undefined)
        errors.push(`Slider missing min/max: ${f.name}`);
    }

    if (f.type === "radio" && !Array.isArray(f.options))
      errors.push(`Radio missing options: ${f.name}`);

    if (f.type === "checklist" && !Array.isArray(f.items))
      errors.push(`Checklist missing items: ${f.name}`);
  });

  return errors;
}
```
